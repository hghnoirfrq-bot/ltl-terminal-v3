        function createMatrixEffect() {
            const matrix = document.getElementById('matrix-bg');
            const characters = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789@#$%^&*()';
            
            for (let i = 0; i < 50; i++) {
                const column = document.createElement('div');
                column.className = 'matrix-column';
                column.style.left = `${Math.random() * 100}%`;
                column.style.animationDuration = `${Math.random() * 5 + 5}s`;
                column.style.animationDelay = `${Math.random() * 5}s`;
                
                let text = '';
                for (let j = 0; j < 20; j++) {
                    text += characters.charAt(Math.floor(Math.random() * characters.length)) + '<br>';
                }
                column.innerHTML = text;
                
                matrix.appendChild(column);
            }
        }
        
        // Check System Status
        async function checkSystemStatus() {
            try {
                const response = await fetch(`${API_URL}/test`);
                const data = await response.json();
                document.getElementById('system-status').innerHTML = 
                    `<span style="color: #00ff00;">ONLINE - Database: ${data.database}</span>`;
            } catch (error) {
                document.getElementById('system-status').innerHTML = 
                    `<span style="color: #ff0000;">OFFLINE - Backend not connected</span>`;
            }
        }
        
        // Load Stats
        async function loadStats() {
            try {
                // Get bookings count
                const bookingsResponse = await fetch(`${API_URL}/bookings`);
                const bookings = await bookingsResponse.json();
                document.getElementById('booking-count').textContent = bookings.length;
                
                // Get users count
                const usersResponse = await fetch(`${API_URL}/users`);
                const users = await usersResponse.json();
                document.getElementById('user-count').textContent = users.length;
                
                // Update visitor count (simulated)
                const visitorCount = parseInt(localStorage.getItem('visitorCount') || '0') + 1;
                localStorage.setItem('visitorCount', visitorCount);
                document.getElementById('visitor-count').textContent = visitorCount;
                
                // Update database status
                document.getElementById('db-status').innerHTML = 
                    '<span style="color: #00ff00;">CONNECTED</span>';
            } catch (error) {
                document.getElementById('db-status').innerHTML = 
                    '<span style="color: #ff0000;">DISCONNECTED</span>';
            }
        }
